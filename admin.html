Step 1 — Set up Google Apps Script

Paste this script into your Google Sheet to receive form submissions

1
Open your BEL Daily Diary Google Sheet
2
Click Extensions → Apps Script
3
Delete all existing code and paste the script below
4
Click Save → then Deploy → New deployment → Web app → set access to Anyone → click Deploy
5
When prompted, click Advanced → Go to project (unsafe) → Allow
6
Copy the Web App URL and paste it into Step 2 below
// BENCHMARK ENGINEERING — DAILY SITE DIARY // Google Apps Script — reads URL parameters (GET request) function doGet(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var p = e.parameter; // Add headers if sheet is empty if (sheet.getLastRow() === 0) { sheet.appendRow([ 'Timestamp','Project','Date','Engineer', 'Planned Workers','Actual Workers','Hours Worked', 'Weather','Weather Impact','Weather Detail', 'Delays','Hours Lost', 'Work Done','Programme Status','Target Met','Target Notes', 'Tomorrow Plan', 'Safety Issues','Safety Detail', 'Materials','Plant', 'Instructions','Other Notes','Sign-Off Name' ]); } sheet.appendRow([ new Date().toLocaleString(), p.project||'', p.date||'', p.engineer||'', p.planned||0, p.actual||0, p.hours||0, p.weather||'', p.weatherImpact||'', p.weatherDetail||'', p.delays||'', p.hoursLost||0, p.workDone||'', p.progStatus||'', p.targetMet||'', p.targetNotes||'', p.tomorrow||'', p.safetyIssues||'', p.safetyDetail||'', p.materials||'', p.plant||'', p.instructions||'', p.otherNotes||'', p.signName||'' ]); return ContentService .createTextOutput('OK') .setMimeType(ContentService.MimeType.TEXT); } catch(err) { return ContentService .createTextOutput('ERROR: ' + err.toString()) .setMimeType(ContentService.MimeType.TEXT); } } function doPost(e) { return doGet(e); }
